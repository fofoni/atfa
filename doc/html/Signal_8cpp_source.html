132 void Signal::filter(Signal imp_resp) {
133  imp_resp.set_samplerate(srate);
134  index_t final_size = samples() + imp_resp.samples() - 1;
135  if (final_size > Signal::DFTDriver::tblsize) {
136  // divide the signal in chunks of size N such that N+K-1==tblsize
137  container_t *big, *small;
138  if (samples() > imp_resp.samples())
139  big = &data, small = &imp_resp.data;
140  else
141  big = &imp_resp.data, small = &data;
142  long N = DFTDriver::tblsize - small->size() + 1;
143  container_t conv(final_size);
144  if (N <= 0) {
145  // fall back to time-domain filtering
146  for (index_t i = 0; i < imp_resp.samples(); i++)
147  for (index_t j = 0; j <= i; j++)
148  conv[i] += data[i-j] * imp_resp[j];
149  for (index_t i = imp_resp.samples(); i < samples(); i++)
150  for (index_t j = 0; j < imp_resp.samples(); j++)
151  conv[i] += data[i-j] * imp_resp[j];
152  for (index_t i = 0; i < conv.size()-samples(); i++)
153  for (index_t j = i+1; j < imp_resp.samples(); j++)
154  conv[samples()+i] += data[samples()+i-j] * imp_resp[j];
155  data = conv; // destroy old data, and copy new from `conv'
156  return;
157  }
158  container_t h_re(DFTDriver::tblsize);
159  container_t h_im(DFTDriver::tblsize);
160  container_t x1(DFTDriver::tblsize);
161  container_t x2(DFTDriver::tblsize);
162  container_t y1(DFTDriver::tblsize);
163  container_t y2(DFTDriver::tblsize);
164  std::copy(small->begin(), small->end(), h_re.begin());
165  dft(h_re, h_im);
166  index_t i;
167  for (i = 0; i != big->size()/(2*N); ++i) {
168  index_t i1 = i*2*N;
169  index_t i2 = i1 + N;
170  std::copy(big->begin() + i1, big->begin() + i1 + N, x1.begin());
171  std::fill(x1.begin() + N, x1.end(), 0);
172  std::copy(big->begin() + i2, big->begin() + i2 + N, x2.begin());
173  std::fill(x2.begin() + N, x2.end(), 0);
174  dft(x1, x2);
175  for (index_t j = 0; j != DFTDriver::tblsize; ++j) {
176  y1[j] = x1[j]*h_re[j] - x2[j]*h_im[j];
177  y2[j] = x1[j]*h_im[j] + x2[j]*h_re[j];
178  }
179  dft(y1, y2, DFTDriver::INVERSE);
180  for (index_t j = 0; j != DFTDriver::tblsize; ++j) {
181  conv[i1 + j] += y1[j];
182  conv[i2 + j] += y2[j];
183  }
184  }
185  {
186  index_t i1 = i*2*N;
187  index_t i2 = i1 + N;
188  long L = big->size() - i1;
189  if (L > N) {
190  std::copy(big->begin() + i1, big->begin() + i1 + N, x1.begin());
191  std::fill(x1.begin() + N, x1.end(), 0);
192  std::copy(big->begin() + i2, big->end(), x2.begin());
193  std::fill(x2.begin() + (L-N), x2.end(), 0);
194  dft(x1, x2);
195  for (index_t j = 0; j != DFTDriver::tblsize; ++j) {
196  y1[j] = x1[j]*h_re[j] - x2[j]*h_im[j];
197  y2[j] = x1[j]*h_im[j] + x2[j]*h_re[j];
198  }
199  dft(y1, y2, DFTDriver::INVERSE);
200  for (index_t j = 0; j != DFTDriver::tblsize; ++j)
201  conv[i1 + j] += y1[j];
202  for (index_t j = 0; j != L-N+small->size()-1; ++j)
203  conv[i2 + j] += y2[j];
204  }
205  else {
206  std::copy(big->begin() + i1, big->end(), x1.begin());
207  std::fill(x1.begin() + L, x1.end(), 0);
208  std::fill(x2.begin(), x2.end(), 0);
209  dft(x1, x2);
210  for (index_t j = 0; j != DFTDriver::tblsize; ++j) {
211  y1[j] = x1[j]*h_re[j] - x2[j]*h_im[j];
212  y2[j] = x1[j]*h_im[j] + x2[j]*h_re[j];
213  }
214  dft(y1, y2, DFTDriver::INVERSE);
215  for (index_t j = 0; j != L+small->size()-1; ++j)
216  conv[i1 + j] += y1[j];
217  }
218  }
219  data = conv; // destroy old data, and copy new from `conv'
220  }
221  else {
222  // fit the whole signal in only one fft
223  index_t L;
224  for (L = DFTDriver::tblsize; L/2 >= final_size; L /= 2) ;
225  // here, final_size fits in L
226  container_t h_re(L);
227  container_t h_im(L);
228  container_t x1(L);
229  container_t x2(L);
230  container_t y1(L);
231  container_t y2(L);
232  std::copy(imp_resp.data.begin(), imp_resp.data.end(), h_re.begin());
233  dft(h_re, h_im);
234  std::copy(data.begin(), data.end(), x1.begin());
235  dft(x1, x2);
236  for (index_t j = 0; j != DFTDriver::tblsize; ++j) {
237  y1[j] = x1[j]*h_re[j] - x2[j]*h_im[j];
238  y2[j] = x1[j]*h_im[j] + x2[j]*h_re[j];
239  }
240  dft(y1, y2, DFTDriver::INVERSE);
241  set_size(final_size);
242  std::copy(y1.begin(), y1.begin() + final_size, data.begin());
243  }
244 }

373  if (srate == 0) {
374  srate = sr;
375  return;
376  }

446 void Signal::DFTDriver::operator ()(container_t& re, container_t& im,
447  const dir_t direction) {
448 
449  if (re.size() != im.size()) {
450  std::ostringstream msg;
451  msg << "Error: Signal::DFTDriver `re' and `im' vectors must be of the "
452  << "same size." << std::endl << " Got: " << re.size()
453  << " and " << im.size() << ".";
454  throw std::runtime_error(msg.str());
455  }
456 
457  index_t L = re.size();
458 
459  if (L > tblsize) {
460  std::ostringstream msg;
461  msg << "Error: DFT of size " << L << ": too big." << std::endl
462  << " Maximum is " << tblsize << ".";
463  throw std::runtime_error(msg.str());
464  }
465 
466  // checks whether n=L is power of two, and also calculates bits=log2(n)
467  { unsigned long n = L;
468  if (L == 0) // nothing to do
469  return;
470  for (bits = 0; n != 1; n /= 2, ++bits)
471  if (n%2 != 0) { // then L is not power of two
472  std::ostringstream msg;
473  msg << "Error: tried to take DFT of vector of size " << L
474  << " (not power of two).";
475  throw std::runtime_error(msg.str());
476  }
477  // here, 2^bits == L
478  }
479 
480  // reverse bits
481  for (unsigned i = 1; i != L-1; ++i) {
482  unsigned j = br(i, bits);
483  if (i >= j) continue;
484  double temp = re[i];
485  re[i] = re[j];
486  re[j] = temp;
487  temp = im[i];
488  im[i] = im[j];
489  im[j] = temp;
490  }
491 
492  for (unsigned size = 1; size != L; size *= 2) {
493  for (unsigned k = 0; k != L/2/size; ++k) {
494  for (unsigned l = 0; l != size; ++l) {
495 
496  // apply single butterfly at indexes i1 and i2,
497  // with coefficient exp(j*tau*frac) at i2.
498  // `frac' is an angle expressed in one-`L'ths of tau,
499  // which is 2*pi.
500 
501  unsigned i1 = 2*size*k + l;
502  unsigned i2 = i1 + size;
503 
504  unsigned frac = L - L/2/size*l; // integer division in 2nd term
505  double cossine = Wre(frac);
506  double sine = direction==DIRECT ? Wim(frac) : -Wim(frac);
507 
508  double tmp_re1 = re[i1];
509  double tmp_im1 = im[i1];
510  double tmp_re2 = re[i2] * cossine - im[i2] * sine;
511  double tmp_im2 = re[i2] * sine + im[i2] * cossine;
512 
513  re[i1] = tmp_re1 + tmp_re2;
514  im[i1] = tmp_im1 + tmp_im2;
515  re[i2] = tmp_re1 - tmp_re2;
516  im[i2] = tmp_im1 - tmp_im2;
517 
518  }
519  }
520  }
521 
522  if (direction==INVERSE)
523  for (unsigned i = 0; i != L; ++i)
524  re[i] /= L, im[i] /= L;
525 
526 }
