130  for (index_t j = 0; j <= i; j++)
131  conv[i] += data[i-j] * imp_resp[j];

136  for (index_t j = i+1; j < imp_resp.samples(); j++)
137  conv[samples()+i] += data[samples()+i-j] * imp_resp[j];
138  data = conv; // destroy old data, and copy new from `conv'
139  return;

147  std::copy(small->begin(), small->end(), h_re.begin());

149  index_t i;

151  index_t i1 = i*2*N;
152  index_t i2 = i1 + N;
153  std::copy(big->begin() + i1, big->begin() + i1 + N, x1.begin());

159  y1[j] = x1[j]*h_re[j] - x2[j]*h_im[j];

177  dft(x1, x2);

180  y2[j] = x1[j]*h_im[j] + x2[j]*h_re[j];
181  }

183  for (index_t j = 0; j != DFTDriver::tblsize; ++j)
184  conv[i1 + j] += y1[j];
185  for (index_t j = 0; j != L-N+small->size()-1; ++j)

428 void Signal::DFTDriver::operator ()(container_t& re, container_t& im,
429  const dir_t direction) {
430 
431  if (re.size() != im.size()) {
432  std::ostringstream msg;
433  msg << "Error: Signal::DFTDriver `re' and `im' vectors must be of the "
434  << "same size." << std::endl << " Got: " << re.size()
435  << " and " << im.size() << ".";
436  throw std::runtime_error(msg.str());
437  }
438 
439  index_t L = re.size();
440 
441  if (L > tblsize) {
442  std::ostringstream msg;
443  msg << "Error: DFT of size " << L << ": too big." << std::endl
444  << " Maximum is " << tblsize << ".";
445  throw std::runtime_error(msg.str());
446  }
447 
448  // checks whether n=L is power of two, and also calculates bits=log2(n)
449  { unsigned long n = L;
450  if (L == 0) // nothing to do
451  return;
452  for (bits = 0; n != 1; n /= 2, ++bits)
453  if (n%2 != 0) { // then L is not power of two
454  std::ostringstream msg;
455  msg << "Error: tried to take DFT of vector of size " << L
456  << " (not power of two).";
457  throw std::runtime_error(msg.str());
458  }
459  // here, 2^bits == L
460  }
461 
462  // reverse bits
463  for (unsigned i = 1; i != L-1; ++i) {
464  unsigned j = br(i, bits);
465  if (i >= j) continue;
466  double temp = re[i];
467  re[i] = re[j];
468  re[j] = temp;
469  temp = im[i];
470  im[i] = im[j];
471  im[j] = temp;
472  }
473 
474  for (unsigned size = 1; size != L; size *= 2) {
475  for (unsigned k = 0; k != L/2/size; ++k) {
476  for (unsigned l = 0; l != size; ++l) {
477 
478  // apply single butterfly at indexes i1 and i2,
479  // with coefficient exp(j*tau*frac) at i2.
480  // `frac' is an angle expressed in one-`L'ths of tau,
481  // which is 2*pi.
482 
483  unsigned i1 = 2*size*k + l;
484  unsigned i2 = i1 + size;
485 
486  unsigned frac = L - L/2/size*l; // integer division in 2nd term
487  double cossine = Wre(frac);
488  double sine = direction==DIRECT ? Wim(frac) : -Wim(frac);
489 
490  double tmp_re1 = re[i1];
491  double tmp_im1 = im[i1];
492  double tmp_re2 = re[i2] * cossine - im[i2] * sine;
493  double tmp_im2 = re[i2] * sine + im[i2] * cossine;
494 
495  re[i1] = tmp_re1 + tmp_re2;
496  im[i1] = tmp_im1 + tmp_im2;
497  re[i2] = tmp_re1 - tmp_re2;
498  im[i2] = tmp_im1 - tmp_im2;
499 
500  }
501  }
502  }
503 
504  if (direction==INVERSE)
505  for (unsigned i = 0; i != L; ++i)
506  re[i] /= L, im[i] /= L;
507 
508 }
